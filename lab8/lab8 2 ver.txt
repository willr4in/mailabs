#include <stdio.h>
#include <math.h>

int abs(int a) {
	if (a > 0) {
		return a;
	}
	else if (a == 0) {
		return 0;
	}
	else {
		return -a;
	}
}

int min(int a, int b) {
	if (a < b) {
		return a;
	}
	else {
		return b;
	}
}

int main() {
	unsigned int n = 1, x = 1, y = 1, m = 0, u = 0, c = 0, d = pow(10, 8);
	scanf_s("%d", &n);
	while (n != 0) {
		scanf_s("%d", &x);
		scanf_s("%d", &y);
		if (x > y) {
			m = m + x;
		}
		else {
			u = u + y;
		}
		c = m + u;
		n--;
		if (abs(x - y) % 5 != 0) {
			d = min(d, abs(x - y));
		}
	}
	if (c % 5 != 0) {
		printf("%d", c);
	}
	else {
		printf("%d", c - d);
	}
	return 0;
}
